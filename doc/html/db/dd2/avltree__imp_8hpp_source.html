667template <class T>
668void AVLTree<T>::remove()
669{
670 // check preconditions.
671 assert(current_exists());
672
673#ifndef NDEBUG
674 // the invariants only must be checked for the first recursive call.
675 // We use a static variable to count the recursion levels.
676 // see section "Static variables in a Function" in
677 // ref https://www.geeksforgeeks.org/static-keyword-cpp/ for more info.
678 static int recursion_count = 0;
679 recursion_count++;
680 if (recursion_count == 1)
681 {
682 // Check invariants.
683 assert(is_a_binary_search_tree());
684 assert(is_a_balanced_tree());
685 }
686#endif // NDEBUG
687
688 bool replace_with_subtree = true;
689 typename AVLTNode<T>::Ref subtree;
690
691 // TODO
692 // Check which of cases 0,1,2,3 we have (see theorical class slides).
693
694 if (curr_->left() == nullptr && curr_->right() == nullptr){
695 subtree = nullptr;
696
697 }else if (curr_->right() == nullptr){
698 subtree = curr_->left();
699
700 }else if (curr_->left() == nullptr){
701 subtree = curr_->right();
702
703 }else{
704 replace_with_subtree = false;
705 }
706
707 //
708
709 if (replace_with_subtree)
710 {
711 // TODO
712 // Manage cases 0,1,2
713 // Remember: update subtree to parent links too.
714
715 if (parent_ == nullptr){
716 root_ = subtree;
717
718 }else if (parent_->right() == curr_){
719 parent_->set_right(subtree);
720
721 if (subtree != nullptr){
722 parent_->right()->set_parent(parent_);
723 }
724
725 }else{
726 parent_->set_left(subtree);
727
728 if (subtree != nullptr){
729 parent_->left()->set_parent(parent_);
730 }
731 }
732
733 curr_ = nullptr;
734
735 //
736 assert(check_parent_chains());
737 make_balanced();
738 assert(check_parent_chains());
739 }
740 else
741 {
742 // TODO
743 // Manage case 3.
744
745 auto tmp = curr_;
746
747 find_inorder_sucessor();
748
749 tmp->set_item(curr_->item());
750
751 remove();
752
753 //
754 }
755
756#ifndef NDEBUG
757 // We come back so the recursion count must be decreased.
758 recursion_count--;
759 assert(recursion_count >= 0);
760 if (recursion_count == 0)
761 {
762 // Only check for the last return.
763 // Check invariants.
764 assert(is_a_binary_search_tree());
765 assert(is_a_balanced_tree());

947template <class T>
948bool AVLTree<T>::check_parent_chains() const
949{
950 if (!is_empty())
951 {
952 std::function<void(typename AVLTNode<T>::Ref, std::vector<T>)> go_down;
953 go_down = [&go_down](typename AVLTNode<T>::Ref node, std::vector<T> branch) -> void
954 {
955 if (node->left() != nullptr || node->right() != nullptr)
956 {
957 branch.push_back(node->item());
958 if (node->left())
959 // go down by the left
960 go_down(node->left(), branch);
961 if (node->right())
962 // go down by the right
963 go_down(node->right(), branch);
964 }
965 else
966 {
967 // The node is a leaf node, so check the branch
968 // to the tree root node.
969 typename AVLTNode<T>::Ref parent = node->parent();
970 int idx = static_cast<int>(branch.size()) - 1;
971 while (node->parent() && idx >= 0)
972 {
973 assert(parent->item() == branch[idx]);
974 --idx;
975 parent = parent->parent();
976 }
977 assert(idx == -1 && parent == nullptr);
978 }
979 };
980 std::vector<T> branch;
981 go_down(root_node(), branch);
982 }
983 return true;
