752template <class T>
753bool AVLTree<T>::check_parent_chains() const
754{
755 if (!is_empty())
756 {
757 std::function<void(typename AVLTNode<T>::Ref, std::vector<T>)> go_down;
758 go_down = [&go_down](typename AVLTNode<T>::Ref node, std::vector<T> branch) -> void
759 {
760 if (node->left() != nullptr || node->right() != nullptr)
761 {
762 branch.push_back(node->item());
763 if (node->left())
764 // go down by the left
765 go_down(node->left(), branch);
766 if (node->right())
767 // go down by the right
768 go_down(node->right(), branch);
769 }
770 else
771 {
772 // The node is a leaf node, so check the branch
773 // to the tree root node.
774 typename AVLTNode<T>::Ref parent = node->parent();
775 int idx = static_cast<int>(branch.size()) - 1;
776 while (node->parent() && idx >= 0)
777 {
778 assert(parent->item() == branch[idx]);
779 --idx;
780 parent = parent->parent();
781 }
782 assert(idx == -1 && parent == nullptr);
783 }
784 };
785 std::vector<T> branch;
786 go_down(root_node(), branch);
787 }
788 return true;
